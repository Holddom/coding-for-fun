**一些基本的string操作**

~~~c++
#include <iostream>
#include <string>
using std::string;
using std::cout;
int main()
{
    string s1;   //空串
    string s2 = s1;//等价于下条语句
    string s(s1);
    string s3 = "hiya";  //等价于下条语句
    string ss("hiya");
    string s4(10, 'c');
    cout << ss;
    return 0;
}
~~~
